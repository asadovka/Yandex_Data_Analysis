```python
# –°–æ–∑–¥–∞–Ω–∏–µ –º–∞—Å—Å–∏–≤–∞ —É–Ω–∏–∫–∞–ª—å–Ω—ã—Ö –ø–∞—Ä –∑–Ω–∞—á–µ–Ω–∏–π –¥–∞—Ç –∏ –≥—Ä—É–ø–ø —Ç–µ—Å—Ç–∞

        {
            'date': 'max',
            'group': 'max',
            'transactionId': pd.Series.nunique,
            'visitorId': pd.Series.nunique,
            'revenue': 'sum',

    ].agg(
        {

        }

# –û–±—ä–µ–¥–∏–Ω–µ–Ω–∏–µ –∫—É–º—É–ª—è—Ç–∏–≤–Ω—ã—Ö –¥–∞–Ω–Ω—ã—Ö –≤ –æ–¥–Ω–æ–π —Ç–∞–±–ª–∏—Ü–µ –∏ –ø—Ä–∏—Å–≤–∞–∏–≤–∞–Ω–∏–µ –µ–µ —Å—Ç–æ–ª–±—Ü–∞–º –ø–æ–Ω—è—Ç–Ω—ã—Ö –Ω–∞–∑–≤–∞–Ω–∏–π
```

```python
#–†–∞—Å—á–µ—Ç –∫–æ–ª–∏—á–µ—Å—Ç–≤–∞ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª–µ–π –≤ –≤—ã–±—Ä–∞–Ω–Ω—É—é –¥–∞—Ç—É –≤ –≥—Ä—É–ø–ø–µ A
visitorsADaily = visitors[visitors['group'] == 'A'][['date', 'visitors']]

#–†–∞—Å—á–µ—Ç –∫–æ–ª–∏—á–µ—Å—Ç–≤–∞ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª–µ–π –≤ –≤—ã–±—Ä–∞–Ω–Ω—É—é –¥–∞—Ç—É –≤ –≥—Ä—É–ø–ø–µ B
visitorsBDaily = visitors[visitors['group'] == 'B'][['date', 'visitors']]

#–†–∞—Å—á–µ—Ç –∫–æ–ª–∏—á–µ—Å—Ç–≤–∞ –∑–∞–∫–∞–∑–æ–≤ –≤ –≤—ã–±—Ä–∞–Ω–Ω—É—é –¥–∞—Ç—É –≤ –≥—Ä—É–ø–ø–µ A

ordersBDaily = (
```

```python
ordersACummulative = ordersADaily.apply(

#–†–∞—Å—á–µ—Ç –∫–æ–ª–∏—á–µ—Å—Ç–≤–∞ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª–µ–π –¥–æ –≤—ã–±—Ä–∞–Ω–Ω–æ–π –¥–∞—Ç—ã –≤–∫–ª—é—á–∏—Ç–µ–ª—å–Ω–æ –≤ –≥—Ä—É–ø–ø–µ B
ordersBCummulative = ordersBDaily.apply(
    lambda x: ordersBDaily[ordersBDaily['date'] <= x['date']].agg(
        {'date': 'max', 'ordersPerDateB': 'sum', 'revenuePerDateB': 'sum'}
    ),

).sort_values(by=['date'])
```

#### –ü–æ–¥–≥–æ—Ç–æ–≤–∫–∞ –≤—ã–±–æ—Ä–æ–∫ –¥–ª—è –ø—Ä–æ–≤–µ–¥–µ–Ω–∏—è —Ä–∞—Å—á–µ—Ç–∞ —Å—Ç–∞—Ç–∏—Å—Ç–∏—á–µ—Å–∫–æ–π –∑–Ω–∞—á–∏–º–æ—Å—Ç–∏


```python
#–†–∞—Å—á–µ—Ç —á–∏—Å–ª–∞ —Å–æ–≤–µ—Ä—à–µ–Ω–Ω—ã—Ö –∑–∞–∫–∞–∑–æ–≤ –¥–ª—è –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª–µ–π, –∫–æ—Ç–æ—Ä—ã–µ –∑–∞–∫–∞–∑—ã–≤–∞–ª–∏ —Ö–æ—Ç—è –±—ã 1 —Ä–∞–∑ –¥–ª—è –≥—Ä—É–ø–ø—ã A
ordersByUsersA = (
    orders[orders['group'] == 'A']
    .groupby('visitorId', as_index=False)
    .agg({'transactionId': pd.Series.nunique})
)
ordersByUsersA.columns = ['visitorId', 'orders']


#–†–∞—Å—á–µ—Ç —á–∏—Å–ª–∞ —Å–æ–≤–µ—Ä—à–µ–Ω–Ω—ã—Ö –∑–∞–∫–∞–∑–æ–≤ –¥–ª—è –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª–µ–π, –∫–æ—Ç–æ—Ä—ã–µ –∑–∞–∫–∞–∑—ã–≤–∞–ª–∏ —Ö–æ—Ç—è –±—ã 1 —Ä–∞–∑ –¥–ª—è –≥—Ä—É–ø–ø—ã B
ordersByUsersB = (
    orders[orders['group'] == 'B']
    .groupby('visitorId', as_index=False)
    .agg({'transactionId': pd.Series.nunique})
)
ordersByUsersB.columns = ['visitorId', 'orders']


#–ü–æ–¥–≥–æ—Ç–æ–≤–∫–∞ –≤—ã–±–æ—Ä–∫–∏ –≥—Ä—É–ø–ø—ã A, –≥–¥–µ —á–∏—Å–ª–æ —ç–ª–µ–º–µ–Ω—Ç–æ–≤ ‚Äî —ç—Ç–æ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª–µ–π, —Å—É–º–º–∞ –≤—Å–µ—Ö —ç–ª–µ–º–µ–Ω—Ç–æ–≤ ‚Äî –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –∑–∞–∫–∞–∑–æ–≤
sampleA = pd.concat(
    [
        ordersByUsersA['orders'],
        pd.Series(
            0,
            index=np.arange(
                data['visitorsPerDateA'].sum() - len(ordersByUsersA['orders'])
            ),
            name='orders',
        ),
    ],
    axis=0,
)


#–ü–æ–¥–≥–æ—Ç–æ–≤–∫–∞ –≤—ã–±–æ—Ä–∫–∏ –≥—Ä—É–ø–ø—ã B, –≥–¥–µ —á–∏—Å–ª–æ —ç–ª–µ–º–µ–Ω—Ç–æ–≤ ‚Äî —ç—Ç–æ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª–µ–π, —Å—É–º–º–∞ –≤—Å–µ—Ö —ç–ª–µ–º–µ–Ω—Ç–æ–≤ ‚Äî –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –∑–∞–∫–∞–∑–æ–≤
sampleB = pd.concat(
    [
        ordersByUsersB['orders'],
        pd.Series(
            0,
            index=np.arange(
                data['visitorsPerDateB'].sum() - len(ordersByUsersB['orders'])
            ),
            name='orders',
        ),
    ],
    axis=0,
)
```

#### –ü—Ä–∏–º–µ–Ω–µ–Ω–∏–µ –∫—Ä–∏—Ç–µ—Ä–∏—è –ú–∞–Ω–Ω–∞-–£–∏—Ç–Ω–∏ –¥–ª—è —Ä–∞—Å—á–µ—Ç–∞ —Å—Ç–∞—Ç–∏—Å—Ç–∏—á–µ—Å–∫–æ–π –∑–Ω–∞—á–∏–º–æ—Å—Ç–∏ —Ä–∞–∑–ª–∏—á–∏–π –≤ –∫–æ–Ω–≤–µ—Ä—Å–∏–∏ –º–µ–∂–¥—É –≥—Ä—É–ø–ø–∞–º–∏


–ü—Ä–∏–º–µ–Ω–∏–º –∫—Ä–∏—Ç–µ—Ä–∏–π –ú–∞–Ω–Ω–∞-–£–∏—Ç–Ω–∏ (–¥–≤—É—Å—Ç–æ—Ä–æ–Ω–Ω–∏–π), —Ç–∞–∫ –≤—ã–±–æ—Ä–∫–∞ –Ω–µ–±–æ–ª—å—à–æ–≥–æ —Ä–∞–∑–º–µ—Ä–∞ –∏ —Å–æ–¥–µ—Ä–∂–∏—Ç –≤—ã–±—Ä–æ—Å—ã.

- **–ù—É–ª–µ–≤–∞—è –≥–∏–ø–æ—Ç–µ–∑–∞ (H0):** —Å—Ç–∞—Ç–∏—Å—Ç–∏—á–µ—Å–∫–∏ –∑–Ω–∞—á–∏–º—ã—Ö —Ä–∞–∑–ª–∏—á–∏–π –≤ –∫–æ–Ω–≤–µ—Ä—Å–∏–∏ –º–µ–∂–¥—É –≥—Ä—É–ø–ø–∞–º–∏ A –∏ B –Ω–µ—Ç. 

- **–ê–ª—å—Ç–µ—Ä–Ω–∞—Ç–∏–≤–Ω–∞—è –≥–∏–ø–æ—Ç–µ–∑–∞ (H1):** —Ä–∞–∑–Ω–∏—Ü–∞ –º–µ–∂–¥—É –∫–æ–Ω–≤–µ—Ä—Å–∏—è–º–∏ –≥—Ä—É–ø–ø A –∏ B —Å—Ç–∞—Ç–∏—Å—Ç–∏—á–µ—Å–∫–∏ –∑–Ω–∞—á–∏–º–∞. 


–ü—Ä–∏–Ω–∏–º–∞–µ–º –∞–ª—å—Ç–µ—Ä–Ω–∞—Ç–∏–≤–Ω—É—é –≥–∏–ø–æ—Ç–µ–∑—É, –µ—Å–ª–∏ p-value –º–µ–Ω—å—à–µ 0.05 (–∫—Ä–∏—Ç–∏—á–µ—Å–∫–∏–π —É—Ä–æ–≤–µ–Ω—å —Å—Ç–∞—Ç–∏—Å—Ç–∏—á–µ—Å–∫–æ–π –∑–Ω–∞—á–∏–º–æ—Å—Ç–∏).


```python
alpha=0.05
p_value = stats.mannwhitneyu(sampleA, sampleB, alternative = 'two-sided')[1].round(4)


print('p_value: ', p_value)
if p_value < alpha:

else:
    print('–ù–µ –ø–æ–ª—É—á–∏–ª–æ—Å—å –æ—Ç–≤–µ—Ä–≥–Ω—É—Ç—å –Ω—É–ª–µ–≤—É—é –≥–∏–ø–æ—Ç–µ–∑—É, –≤—ã–≤–æ–¥ –æ —Ä–∞–∑–ª–∏—á–∏–∏ —Å–¥–µ–ª–∞—Ç—å –Ω–µ–ª—å–∑—è') 


print("\n–û—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ–µ –ø—Ä–µ–∏–º—É—â–µ—Å—Ç–≤–æ –ø–æ –∫–æ–Ω–≤–µ—Ä—Å–∏–∏ –≤ –≥—Ä—É–ø–ø–µ B: {0:.2%}".format(
    (
        data['ordersPerDateB'].sum()/data['visitorsPerDateB'].sum()
    )

     )
```
